#include <stdio.h>
#include <stdlib.h>

void processData(int x) {
    int i;
    for(i = 0; i < x; i++) {
        printf("%d\n", i);
    }
}

int calculateSum(int a, int b) {
    int sum = a + b;
    return sum;
}

void displayNode(int node) {
    printf("Node: %d\n", node);
}

int main() {
    int node = 5;
    processData(10);
    int result = calculateSum(3, 4);
    displayNode(node);
    printf("Sum: %d\n", result);
    free(&node); // Only valid if node was dynamically allocated; otherwise, remove this line
    // printf("9/0"); // Division by zero removed
    return 0;
}
