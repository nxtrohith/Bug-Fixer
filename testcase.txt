#include <stdio.h>
#include <stdlib.h>

void processData(int x) {
    int i;
    for(i = 0; i < x; i++) {
        printf("%d\n", i);
    }
}

int calculateSum(int a, int b) {
    int sum = a + b;
    return sum;
}

void displayNode(int node) {
    printf("Node: %d\n", node);
}

int main() {
    // Dynamic allocation
    int *node = malloc(sizeof(int));

        return 1;
    }
    *node = 5;

    int result = calculateSum(3, 4);
